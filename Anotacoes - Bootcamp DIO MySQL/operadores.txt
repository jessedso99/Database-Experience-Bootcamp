distinct - mostra apenas uma ocorrencia de cada valor da coluna
	SELECT DISTINCT salary FROM employee;

operadores booleanos = (AND, OR, XOR, NOT)
	AND - todos tem q ser igual
	OR - ou um ou outro
	XOR - ou exclusivo, pesquisar.
	NOT - nega. Ex: not in (1,2,4) = 0, 3, 5, ...
